Option Explicit
'le module pour afficher la position du contre-Torpilleur que le joueur2 selecte sur le worksheet Joueur2
Sub Bateau2_3_Sub()
    Dim Range As Range
    Dim Question As Variant
    Set Range = Selection
    If Range.Count = 3 Then
        Question = MsgBox("Est-ce que vous voulez placer Contre-torpilleur ici?", vbQuestion + vbYesNoCancel, "Place du Contre-torpilleur")
        If Question = vbYes Then
            If Total_2 < 5 Then
                Selection.Interior.ColorIndex = 4
                Selection.Name = "Bateau3"
                Total_2 = Total_2 + 1
                Total2_3 = Total2_3 + 1
                
                If Total2_3 = 1 Then
                    Call remplir_cordonnee_bateau(bateau31_2)
                    Call ajouter_bateau_a_carte(bateau31_2, Ocean_Joueur2, 3)
                ElseIf Total2_3 = 2 Then
                    Call remplir_cordonnee_bateau(bateau32_2)
                    Call ajouter_bateau_a_carte(bateau32_2, Ocean_Joueur2, 3)
                End If
                
            End If
        ElseIf Question = vbNo Then
            Exit Sub
        Else
            PlacerBateau_Joueur2.Show
        End If
    End If
End Sub

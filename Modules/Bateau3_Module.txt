Option Explicit
'le module pour afficher la position du contre-Torpilleur que le joueur1 selecte sur le worksheet Joueur1
Sub Bateau3_Sub()
    Dim Range As Range
    Dim Question As Variant
    Set Range = Selection
    If Range.Count = 3 Then
        Question = MsgBox("Est-ce que vous voulez placer le contre-torpilleur ici?", vbQuestion + vbYesNoCancel, "Place du Contre-torpilleur")
        If Question = vbYes Then
            If Total < 5 Then
                Selection.Interior.ColorIndex = 4
                Selection.Name = "Bateau3"
                Total = Total + 1
                Total3 = Total3 + 1
                If Total3 = 1 Then
                    Call remplir_cordonnee_bateau(bateau31_1)
                    Call ajouter_bateau_a_carte(bateau31_1, Ocean_Joueur1, 3)
                ElseIf Total3 = 2 Then
                    Call remplir_cordonnee_bateau(bateau32_1)
                    Call ajouter_bateau_a_carte(bateau32_1, Ocean_Joueur1, 3)
                End If
            End If
        ElseIf Question = vbNo Then
            Exit Sub
        Else
            PlacerBateau_Joueur1.Show
        End If
    End If
End Sub

Option Explicit


'le module pour afficher la position du Porte_avions que le joueur1 selecte sur le worksheet Joueur1


Sub Bateau5_Sub()
    Dim Range As Range
    Dim Question As Variant
    Set Range = Selection
    If Range.Count = 5 Then
        Question = MsgBox("Est-ce que vous voulez placer Porte_avions ici?", vbQuestion + vbYesNoCancel, "Place de la Porte avions")
        If Question = vbYes Then
            If Total2 < 5 Then
                Selection.Interior.ColorIndex = 4
                Selection.Name = "Bateaude5"
                Total = Total2 + 1
                Total5 = Total5 + 1
                
                Call remplir_cordonnee_bateau(bateau5_1)
                Call ajouter_bateau_a_carte(bateau5_1, Ocean_Joueur1, 5)
                
            End If
        ElseIf Question = vbNo Then
            Exit Sub
        Else
            PlacerBateau_Joueur1.Show
        End If
    End If
            
End Sub

Option Explicit
'le module pour afficher la position du Torpilleur que le joueur1 selecte sur le worksheet Joueur1
Sub Bateau2_Sub()
    Dim Range As Range
    Dim Question As Variant
    Set Range = Selection
    If Range.Count = 2 Then
        Question = MsgBox("Est-ce que vous voulez placer Torpilleur?", vbQuestion + vbYesNoCancel, "Place du Torpilleur")
        If Question = vbYes Then
            If Total < 5 Then
                Selection.Interior.ColorIndex = 4
                Selection.Name = "Bateau2"
                Total = Total + 1
                Total2 = Total2 + 1
                
                Call remplir_cordonnee_bateau(bateau2_1)
                Call ajouter_bateau_a_carte(bateau2_1, Ocean_Joueur1, 2)
                
            End If
        ElseIf Question = vbNo Then
            Exit Sub
        Else
            PlacerBateau_Joueur1.Show
        End If
    End If
End Sub

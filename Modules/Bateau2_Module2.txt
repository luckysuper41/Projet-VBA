Option Explicit
'le module pour afficher la position du Torpilleur que le joueur1 selecte sur le worksheet Joueur2
Sub Bateau2_2_Sub()
    Dim Range As Range
    Dim Question As Variant
    Set Range = Selection
    If Range.Count = 2 Then
        Question = MsgBox("Est-ce que vous voulez placer torpilleur ici?", vbQuestion + vbYesNoCancel, "Place du torpilleur")
        If Question = vbYes Then
            If Total_2 < 5 Then
                Selection.Interior.ColorIndex = 4
                Selection.Name = "Bateau2"
                Total_2 = Total_2 + 1
                Total2_2 = Total2_2 + 1
                
                Call remplir_cordonnee_bateau(bateau2_2)
                Call ajouter_bateau_a_carte(bateau2_2, Ocean_Joueur2, 2)
                
            End If
        ElseIf Question = vbNo Then
            Exit Sub
        Else
            PlacerBateau_Joueur2.Show
        End If
    End If
End Sub
